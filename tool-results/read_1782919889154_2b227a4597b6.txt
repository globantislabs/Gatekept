   103→  return (
   104→    <motion.section
   105→      ref={ref}
   106→      id={id}
   107→      initial="hidden"
   108→      animate={isInView ? 'visible' : 'hidden'}
   109→      variants={staggerContainer}
   110→      className={className}
   111→    >
   112→      {children}
   113→    </motion.section>
   114→  )

   118→// COUNTER ANIMATION COMPONENT
   119→// ============================================================
   120→function AnimatedCounter({ end, suffix = '', prefix = '' }: { end: number; suffix?: string; prefix?: string }) {
   121→  const [count, setCount] = useState(0)
   122→  const ref = useRef(null)
   123→  const isInView = useInView(ref, { once: true })
   124→  useEffect(() => {
   125→    if (!isInView) return
   126→    let start = 0
   127→    const duration = 2000
   128→    const increment = end / (duration / 16)
   129→    const timer = setInterval(() => {
   130→      start += increment
   131→      if (start >= end) { setCount(end); clearInterval(timer) }
   132→      else setCount(Math.floor(start))
   133→    }, 16)
   134→    return () => clearInterval(timer)
   135→  }, [isInView, end])
   136→  return <span ref={ref}>{prefix}{count.toLocaleString()}{suffix}</span>
   137→}
