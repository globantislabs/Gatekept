   127→function AnimatedCounter({ end, suffix = '', prefix = '' }: { end: number; suffix?: string; prefix?: string }) {
   128→  const [count, setCount] = useState(0)
   129→  const ref = useRef(null)
   130→  const isInView = useInView(ref, { once: true })
   131→  useEffect(() => {
   132→    if (!isInView) return
   133→    let start = 0
   134→    const duration = 2000
   135→    const increment = end / (duration / 16)
   136→    const timer = setInterval(() => {
   137→      start += increment
   138→      if (start >= end) { setCount(end); clearInterval(timer) }
   139→      else setCount(Math.floor(start))
   140→    }, 16)
   141→    return () => clearInterval(timer)
   142→  }, [isInView, end])
   143→  return <span ref={ref}>{prefix}{count.toLocaleString()}{suffix}</span>
   144→}
